    <script>
        // State
        let dados = JSON.parse(localStorage.getItem('motorista_pro_ultra_v1')) || { ganhos: [], abastecimentos: [], despesas: [] };
        let myChart;

        // Dark Mode
        function toggleDarkMode() {
            document.body.classList.toggle('dark');
            const isDark = document.body.classList.contains('dark');
            document.getElementById('dark-icon').innerText = isDark ? '‚òÄÔ∏è' : 'üåô';
            localStorage.setItem('darkMode', isDark);
            if(myChart) initChart(); // Redraw chart for colors
        }

        // Check Preference
        if(localStorage.getItem('darkMode') === 'true') toggleDarkMode();

        function openTab(tabId) {
            document.querySelectorAll('.tab-content').forEach(tab => tab.classList.remove('active'));
            document.getElementById(tabId).classList.add('active');
            document.querySelectorAll('nav button').forEach(btn => { 
                btn.classList.remove('active'); 
                btn.classList.add('text-slate-400'); 
            });
            if(event && event.currentTarget) {
                event.currentTarget.classList.add('active');
                event.currentTarget.classList.remove('text-slate-400');
            }
            atualizarDashboard();
        }

        function formatBRL(val) { return val.toLocaleString('pt-BR', { style: 'currency', currency: 'BRL' }); }

        function salvarEntrada() {
            const valor = parseFloat(document.getElementById('valor-ganho').value);
            const desc = document.getElementById('desc-ganho').value || 'Corrida';
            if(!valor) return;
            dados.ganhos.push({ valor, desc, data: new Date().toISOString() });
            save();
            document.getElementById('valor-ganho').value = '';
        }

        function salvarAbastecimento() {
            const km = parseFloat(document.getElementById('comb-km-rodado').value);
            const litros = parseFloat(document.getElementById('comb-litros').value);
            const valor = parseFloat(document.getElementById('comb-valor').value);
            if(!km || !litros || !valor) return;
            dados.abastecimentos.push({ km, litros, valor, media: km/litros, data: new Date().toISOString() });
            save();
            document.getElementById('comb-km-rodado').value = '';
            document.getElementById('comb-litros').value = '';
            document.getElementById('comb-valor').value = '';
        }

        function salvarDespesa() {
            const tipo = document.getElementById('tipo-despesa').value;
            const valor = parseFloat(document.getElementById('valor-despesa').value);
            if(!valor) return;
            dados.despesas.push({ tipo, valor, data: new Date().toISOString() });
            save();
            document.getElementById('valor-despesa').value = '';
        }

        function save() {
            localStorage.setItem('motorista_pro_ultra_v1', JSON.stringify(dados));
            atualizarDashboard();
            renderListas();
            initChart();
        }

        // L√≥gica de Gr√°fico
        function initChart() {
            const canvas = document.getElementById('weeklyChart');
            if(!canvas) return;
            const ctx = canvas.getContext('2d');
            if(myChart) myChart.destroy();

            const last7Days = [...Array(7)].map((_, i) => {
                const d = new Date();
                d.setDate(d.getDate() - i);
                return d.toLocaleDateString('pt-BR', { weekday: 'short' });
            }).reverse();

            const dailyTotals = [...Array(7)].map((_, i) => {
                const d = new Date();
                d.setDate(d.getDate() - (6 - i));
                const dayStr = d.toLocaleDateString('pt-BR');
                return dados.ganhos
                    .filter(g => new Date(g.data).toLocaleDateString('pt-BR') === dayStr)
                    .reduce((a, b) => a + b.valor, 0);
            });

            const isDark = document.body.classList.contains('dark');

            myChart = new Chart(ctx, {
                type: 'bar',
                data: {
                    labels: last7Days,
                    datasets: [{
                        label: 'Ganhos (R$)',
                        data: dailyTotals,
                        backgroundColor: '#6366f1',
                        borderRadius: 8,
                        barThickness: 20
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: { legend: { display: false } },
                    scales: {
                        y: { display: false },
                        x: { 
                            grid: { display: false },
                            ticks: { color: isDark ? '#94a3b8' : '#64748b', font: { weight: 'bold', size: 10 } }
                        }
                    }
                }
            });
        }

        // Exporta√ß√£o CSV
        function exportarDados() {
            let csv = "Tipo,Descricao,Valor,Data\n";
            dados.ganhos.forEach(g => csv += `Ganho,${g.desc},${g.valor},${new Date(g.data).toLocaleString()}\n`);
            dados.despesas.forEach(d => csv += `Despesa,${d.tipo},${d.valor},${new Date(d.data).toLocaleString()}\n`);
            dados.abastecimentos.forEach(a => csv += `Abastecimento,${a.km}km,${a.valor},${new Date(a.data).toLocaleString()}\n`);

            const blob = new Blob([csv], { type: 'text/csv' });
            const url = window.URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.setAttribute('hidden', '');
            a.setAttribute('href', url);
            a.setAttribute('download', `relatorio_motorista_${new Date().toLocaleDateString()}.csv`);
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
        }

        function atualizarDashboard() {
            const totalG = dados.ganhos.reduce((a, b) => a + b.valor, 0);
            const totalA = dados.abastecimentos.reduce((a, b) => a + b.valor, 0);
            const totalD = dados.despesas.reduce((a, b) => a + b.valor, 0);
            const totalS = totalA + totalD;
            
            document.getElementById('total-ganhos').innerText = formatBRL(totalG);
            document.getElementById('total-saidas').innerText = formatBRL(totalS);
            document.getElementById('saldo-final').innerText = formatBRL(totalG - totalS);

            const mediaFinal = dados.abastecimentos.length > 0 ? dados.abastecimentos.reduce((a, b) => a + b.media, 0) / dados.abastecimentos.length : 0;
            document.getElementById('media-kml').innerText = mediaFinal.toFixed(2);

            const categorias = { 
                'alimentacao': 'üç± Alimenta√ß√£o', 
                'manutencao': 'üîß Manuten√ß√£o', 
                'emergencia': 'üÜò Seguran√ßa', 
                'investimento': 'üìà Metas',
                'futeis': 'üé≠ Futilidades'
            };

            document.getElementById('container-reservas').innerHTML = Object.entries(categorias).map(([slug, label]) => {
                const total = dados.despesas.filter(d => d.tipo === slug).reduce((a, b) => a + b.valor, 0);
                return `<div class="bg-white dark:bg-slate-800 p-5 rounded-3xl border border-slate-100 dark:border-slate-700 flex justify-between items-center"><div class="flex items-center gap-4"><div class="w-10 h-10 bg-slate-50 dark:bg-slate-700 rounded-2xl flex items-center justify-center text-sm shadow-sm">${label.split(' ')[0]}</div><div><p class="text-[10px] font-black text-slate-400 uppercase leading-none mb-1">${label.split(' ')[1]}</p><p class="text-base font-black tracking-tight">${formatBRL(total)}</p></div></div></div>`;
            }).join('');
        }

        function renderListas() {
            const listG = document.getElementById('lista-ganhos');
            if(listG) listG.innerHTML = dados.ganhos.slice(-3).reverse().map(g => `<div class="p-5 bg-white dark:bg-slate-800 rounded-3xl border border-slate-100 dark:border-slate-700 flex justify-between items-center"><div><p class="text-[11px] font-black">${g.desc}</p><p class="text-[9px] text-slate-400 font-bold uppercase">${new Date(g.data).toLocaleDateString()}</p></div><span class="text-emerald-500 font-black text-sm">+ ${formatBRL(g.valor)}</span></div>`).join('');
            
            const listA = document.getElementById('lista-abastecimento');
            if(listA) listA.innerHTML = dados.abastecimentos.slice(-3).reverse().map(a => `<div class="p-5 bg-white dark:bg-slate-800 rounded-3xl border border-slate-100 dark:border-slate-700 flex justify-between items-center"><div><p class="text-[11px] font-black">${a.km}km rodados</p><p class="text-[9px] text-slate-400 font-bold uppercase">${new Date(a.data).toLocaleDateString()} ‚Ä¢ ${a.litros}L</p></div><span class="text-indigo-600 font-black text-sm">${a.media.toFixed(2)} km/L</span></div>`).join('');
        }

        window.onload = () => {
            save();
            initChart();
        };
    </script>
